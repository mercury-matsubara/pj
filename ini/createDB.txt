DROP DATABASE project;
COMMIT;
CREATE DATABASE project CHARACTER SET cp932;
COMMIT;
USE project;
create table projectnuminfo
( 
1CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (1CODE),
PROJECTNUM VARCHAR (8) NOT NULL UNIQUE ,
PROJECTNAME VARCHAR (60) NOT NULL 
);

create table edabaninfo
( 
2CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (2CODE),
EDABAN VARCHAR (8) NOT NULL UNIQUE ,
PJNAME VARCHAR (60) NOT NULL 
);

create table kouteiinfo
( 
3CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (3CODE),
KOUTEIID VARCHAR (6) NOT NULL UNIQUE ,
KOUTEINAME VARCHAR (60) NOT NULL 
);

create table syaininfo
( 
4CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (4CODE),
STAFFID VARCHAR (8) NOT NULL UNIQUE ,
STAFFNAME VARCHAR (60) NOT NULL 
);

create table projectinfo
( 
5CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (5CODE),
1CODE BIGINT (8) NOT NULL ,
2CODE BIGINT (8) NOT NULL ,
CHARGE BIGINT (8) NOT NULL ,
5ENDDATE DATE ,
5PJSTAT VARCHAR (1) DEFAULT '1' 
);

create table projectditealinfo
( 
6CODE BIGINT (8) NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (6CODE),
4CODE BIGINT (8) NOT NULL ,
5CODE BIGINT (8) NOT NULL ,
DETALECHARGE BIGINT (8) NOT NULL ,
5PK BIGINT (8) ,
6ENDDATE DATE ,
6PJSTAT VARCHAR (1) DEFAULT '1' 
);

create table progressinfo
( 
7CODE BIGINT (8) NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (7CODE),
3CODE BIGINT (8) NOT NULL ,
6CODE BIGINT (8) NOT NULL ,
SAGYOUDATE DATE NOT NULL ,
TEIZITIME DOUBLE (3,2) NOT NULL ,
ZANGYOUTIME DOUBLE (3,2) NOT NULL ,
7ENDDATE DATE ,
7PJSTAT VARCHAR (1) DEFAULT '1' 
);

create table endpjinfo
( 
8CODE BIGINT (8) NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (8CODE),
6CODE BIGINT (8) NOT NULL ,
TEIJITIME DOUBLE (4,2) NOT NULL ,
ZANGYOTIME DOUBLE (4,2) NOT NULL ,
TOTALTIME DOUBLE (4,2) NOT NULL ,
PERFORMANCE DOUBLE (6,3) NOT NULL ,
8ENDDATE DATE ,
8PJSTAT VARCHAR (1) DEFAULT '1' 
);

create table monthdatainfo
( 
9CODE BIGINT (8) NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (9CODE),
4CODE BIGINT (8) ,
5CODE BIGINT (8) ,
PERIOD VARCHAR (4) ,
MONTH VARCHAR (2) ,
ITEM VARCHAR (40) ,
VALUE VARCHAR (30) ,
9ENDDATE DATE 
);

create table loginuserinfo 
(
LUSERID BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
LUSERNAME VARCHAR (50) NOT NULL ,
LUSERPASS VARCHAR (50) NOT NULL );
create table systeminfo 
(
ORNERID BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
STARTDATE DATE NOT NULL );
COMMIT;
